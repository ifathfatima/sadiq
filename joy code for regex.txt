package com.cg.validations;

import java.util.regex.Pattern;

public class Validations {
public boolean checkName(String Name)
{
if(Pattern.matches("[A-Z][a-z]{2,}", Name)) 
{
return true;
}
else
{
return false;
}

}
public boolean checkDob(String dob)
{
if(Pattern.matches("[0-9]{2}[/][0-9]{2}[/][0-9]{4}", dob)) 
{
return true;
}
else
{
return false;
}

}
public boolean checkEmailId(String emailid)
{
if(Pattern.matches("[0-9a-zA-Z][0-9a-zA-Z._$]{1,}[@][0-9a-zA-Z]{1,}[.][a-zA-Z]{1,}", emailid)) 
{
return true;
}
else
{
return false;
}

}

{

package com.cp.dio;

import java.util.Scanner;

import com.cg.bean.Bean;
import com.cg.validations.Validations;

public class EnterDetails {

public static void main(String[] args) {
// TODO Auto-generated method stub
Bean obj=new Bean();
Validations v=new Validations();
Scanner sc=new Scanner(System.in);
System.out.println("Enter Your Details");
System.out.println("__________________");
System.out.println("Enter First name(Only First Letter Capitalized):");
obj.setName(sc.next());
if(!(v.checkName(obj.getName())))
{
System.out.println("Please Enter a valid name with only first letter in capitals!");
System.exit(0);
}
System.out.println("Enter Date of Birth(in dd/mm/yyyy format):");
obj.setDob(sc.next());
if(!(v.checkDob(obj.getDob())))
{
System.out.println("Please Enter a valid date of birth in dd/mm/yyyy format!");
System.exit(0);
}
System.out.println("Enter PhoneNumber:");
obj.setPhonenumber(sc.next());
if(!(v.checkPhoneNumber(obj.getPhonenumber())))
{
System.out.println("Please Enter a valid 10 digit phonenumber!");
System.exit(0);
}
System.out.println("Enter email id:");
obj.setEmailid(sc.next());
if(!(v.checkEmailId(obj.getEmailid())))
{
System.out.println("Please Enter a valid email id!");
System.exit(0);
}
sc.close();
System.out.println("Successfully Captured your information!");
}

}
